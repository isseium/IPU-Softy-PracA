\section{三角関数\cite{yosshy}}
\begin{itemize}
\item 2倍角の公式\\

\begin{eqnarray*}
\sin 2\alpha & = &2\sin\theta\cos\theta\\
\cos 2\alpha & = & \cos^{2} \theta - \sin^{2} \theta\\
	& = & 2 \cos^{2}\theta - 1\\
	& = & 1 - 2\sin^{2} \theta\\
\end{eqnarray*}

\newpage

\item  5倍角の公式\\
\begin{eqnarray*}
\sin 5\theta & = & 16 \sin^{5}\theta - 20\sin^{3}\theta + 5 \sin \theta\\
\cos 5\theta & = & 16 \cos^{5}\theta - 20\cos^{3}\theta + 5 \cos \theta\\
\end{eqnarray*}

\item $36^\circ$ の正弦、余弦\\
\begin{eqnarray*}
\sin 36^\circ & = & \cos 54^\circ\\
	& = & \sqrt{\frac{5-\sqrt{5}}{8}}\\
\cos 36^\circ & = & \sin 54^\circ\\
	& = & \frac{\sqrt{5}+1}{4}\\
\end{eqnarray*}

\item $72^\circ$ の正弦、余弦\\
\begin{eqnarray*}
\sin 72^\circ & = & \sqrt{\frac{5+\sqrt{5}}{8}}\\

\end{eqnarray*}
\end{itemize}

\section{統計学の公式\cite{wiki}}
\begin{itemize}
\item 算術平均値\\
\[
\bar{X} = \frac{1}{n}\sum_{i=1}^{n}X_{i}
\]

\item 分散\\
\[
\sigma^2 = \frac{1}{n}\sum_{i=1}^{n}(X_{i}-\bar{X})^2
\]

\item 標準偏差\\
\[
SD = \sigma = \sqrt{\frac{1}{n}\sum_{i=1}^{n}(X_{i}-\bar{X})^2}
\]
\end{itemize}

\section{加法定理}
\begin{eqnarray*}
\sin(\alpha+\beta) & = & \sin\alpha\cos\beta+\cos\alpha\sin\beta\\
\sin(\alpha-\beta) & = & \sin\alpha\cos\beta-\cos\alpha\sin\beta\\
\cos(\alpha+\beta) & = & \cos\alpha\cos\beta-\sin\alpha\sin\beta\\
\cos(\alpha-\beta) & = & \cos\alpha\cos\beta+\sin\alpha\sin\beta\\
\tan(\alpha+\beta) & = & \frac{\tan\alpha+\tan\beta}{1-\tan\alpha\tan\beta}\\
\tan(\alpha-\beta) & = & \frac{\tan\alpha-\tan\beta}{1+\tan\alpha\tan\beta}\\
\end{eqnarray*}

\end{eqnarray*}

\begin{eqnarray*}
\theta & = & 30^{\circ},150^{\circ}\\
\end{eqnarray*}

(b)
\begin{eqnarray*}
\frac{dy}{dx} & = & \{(2x^{2}-1)(x+3)\}'\\
	& = & \{x^3+3x^2+x-3\}'\\
	& = & 3x^2 + 6x - 1\\
\end{eqnarray*}

(c)
\begin{eqnarray*}
\int_{0}^{1} (x^3+1)dx & = & \left[\frac{1}{4}x^4+x\right]_{0}^{1}\\
